% ### Definición de terminología nueva.
% Uso: \terminology[entrada-índice]{Término}
%   - El argumento obligatorio {Término} es lo que se imprime en el texto (negrita).
%   - El argumento opcional [entrada-índice] controla cómo se ordena en el índice alfabético.
%   - Si se omite, el término se usa también como entrada del índice.
% Se usa \NewDocumentCommand (LaTeX3) en lugar de \newcommand porque
% \terminology aparece dentro de \item[...], y xparse maneja correctamente
% el anidamiento de corchetes.
\NewDocumentCommand{\terminology}{om}{%
  \IfNoValueTF{#1}
  {\index{#2}{\boldmath \bfseries #2}}
  {\index{#1}{\boldmath \bfseries #2}}%
}

% Comando para referenciar personas en el texto y el índice
% \person{Aristóteles} → índice: “Aristóteles”
% \person[Bacon]{Francis} → typeset: “Francis Bacon”, índice: “Bacon, Francis”
\newcommand{\person}[2][]{%
  \ifx\relax#1\relax
    #2\index{#2}%
  \else
    #2 #1\index{#1, #2}%
  \fi}

% Radio de esquinas redondeadas para entornos con color de fondo
\newcommand{\bgboxradius}{6pt}
\newcommand{\bgboxpadding}{7pt}

% ### Entorno para hacer divagaciones 'digress'
% Flota como una figura para evitar saltos de página incómodos,
% pero sin interferir con la numeración de figuras reales.
\newcommand{\digresswidth}{0.985\textwidth}
\colorlet{digresscolor}{Linen}
\newsavebox{\digressbox}
% Crear un tipo de flotante dedicado para las digresiones
\newfloat{digressfloat}{htbp}{}
\newenvironment{digress}[1]{
  \begin{digressfloat}
    \centering
    \begin{lrbox}{\digressbox}\begin{minipage}{\digresswidth}
        \setlength{\parindent}{\normalparindent}%
        {\subsubsection*{{\normalfont\emoji{mag}} #1}}}
        {\end{minipage}\end{lrbox}
    \medskip\par\noindent%
    \tikz[baseline]{\node[fill=digresscolor, rounded corners=\bgboxradius,
        inner sep=\bgboxpadding, draw=LightGray, line width=0.4pt]{\usebox{\digressbox}};}
  \end{digressfloat}}

% ### Entorno para resaltar conceptos importantes 'remember'
% con un emoji de disquete.
\newcommand{\rememberwidth}{0.985\textwidth}
\colorlet{remembercolor}{Lavender}
\newsavebox{\rememberbox}
\newenvironment{remember}{
  \begin{lrbox}{\rememberbox}\begin{minipage}{\rememberwidth}
      \setlength{\parindent}{\normalparindent}%
      {\subsubsection*{{\normalfont\emoji{floppy-disk}} Para recordar}}}
      {\end{minipage}\end{lrbox}
  \bigskip\par\noindent%
  \tikz[baseline]{\node[fill=remembercolor, rounded corners=\bgboxradius,
      inner sep=\bgboxpadding, draw=LightGray, line width=0.4pt]{\usebox{\rememberbox}};}\bigskip}
